// --- 自動再生 (Web Audio API の正確なタイムラインを使用) --- // テスト用メロディ (キラキラ星の冒頭: 拍単位で指定) const demoMelody = [ { note: 'C5', beat: 0 }, { note: 'C5', beat: 1 }, { note: 'G5', beat: 2 }, { note: 'G5', beat: 3 }, { note: 'A5', beat: 4 }, { note: 'A5', beat: 5 }, { note: 'G5', beat: 6 } ]; let scheduledNodes = []; // 自動再生中の音源ノードを管理する配列 let UIAnimationTimers = []; // 鍵盤を光らせるためのタイマー配列 function stopAllMusic() { // 1. Web Audio API側で予約されている音をすべて即座にストップ scheduledNodes.forEach(node => { try { node.osc.stop(); } catch(e) {} }); scheduledNodes = []; // 2. 画面を光らせるためのタイマー(setTimeout)をすべて解除 UIAnimationTimers.forEach(t => clearTimeout(t)); UIAnimationTimers = []; // 3. 手動で鳴らしている音も含めてすべて消音 Object.keys(activeNodes).forEach(note => stopNote(note)); // 4. すべての鍵盤の光るエフェクトをリセット document.querySelectorAll('.key').forEach(keyEl => keyEl.classList.remove('active')); } // ▶ Playボタンのクリックイベント document.getElementById('play-btn').addEventListener('click', () => { stopAllMusic(); // 二重再生を防ぐためにまずリセット initAudio(); // 現在の入力欄から最新のBPMを取得し、1拍あたりの秒数を計算 const bpm = parseFloat(document.getElementById('bpm-input').value) || 120; const secondsPerBeat = 60 / bpm; // 再生を開始する「今」の絶対時間を取得 const startTime = audioCtx.currentTime + 0.1; // 100msのバッファを持たせて滑らかに開始 demoMelody.forEach(item => { if (!noteFreqs[item.note]) return; // 拍数から、この音が鳴るべき正確な「秒数」を計算 const noteStartTime = startTime + (item.beat * secondsPerBeat); const noteStopTime = noteStartTime + (secondsPerBeat * 0.8); // 1拍の80%の長さだけ鳴らす // --- 音源（オシレーター）のスケジュール設定 --- const osc = audioCtx.createOscillator(); const gainNode = audioCtx.createGain(); osc.type = 'triangle'; osc.frequency.setValueAtTime(noteFreqs[item.note], noteStartTime); // 音量のスケジュール（アタックとリリース） gainNode.gain.setValueAtTime(0, startTime); gainNode.gain.setValueAtTime(0, noteStartTime); gainNode.gain.linearRampToValueAtTime(0.4, noteStartTime + 0.02); // 立ち上がり gainNode.gain.setValueAtTime(0.4, noteStopTime); gainNode.gain.exponentialRampToValueAtTime(0.001, noteStopTime + 0.1); // 余韻 osc.connect(gainNode); gainNode.connect(audioCtx.destination); // 正確なWeb Audioのタイムライン上で再生・停止を予約 osc.start(noteStartTime); osc.stop(noteStopTime + 0.1); // ストップボタン用にノードを記憶 scheduledNodes.push({ osc, gainNode }); // --- 画面上の鍵盤をタイミングよく光らせる処理 --- const delayToSyncUI = (noteStartTime - audioCtx.currentTime) * 1000; const delayToStopUI = (noteStopTime - audioCtx.currentTime) * 1000; const keyEl = document.querySelector(`.key[data-note="${item.note}"]`); if (keyEl) { // 音が鳴る瞬間に鍵盤を光らせる const startTimer = setTimeout(() => { keyEl.classList.add('active'); }, delayToSyncUI); // 音が止まる瞬間に鍵盤の光を消す const stopTimer = setTimeout(() => { keyEl.classList.remove('active'); }, delayToStopUI); UIAnimationTimers.push(startTimer, stopTimer); } }); }); // ■ Stopボタンのクリックイベント document.getElementById('stop-btn').addEventListener('click', () => { stopAllMusic(); });
